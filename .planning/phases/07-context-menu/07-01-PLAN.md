    ```typescript
    // ---- Context menu ----
    function closeMenu() {
      menuVisible.value = false

    function onContextMenu(e: MouseEvent) {
      // Right-click on result rows is reserved for future per-result menu — ignore
      if ((e.target as HTMLElement).closest('.result-row')) return
      // Clamp X so menu (min-width 160px) does not overflow the 500px window
      menuX.value = Math.min(e.clientX, 500 - 170)
      menuY.value = e.clientY
      menuVisible.value = true
    }

    async function openSettings() {
      closeMenu()
      await invoke('open_settings_window').catch(console.error)
    }

    async function quitApp() {
      closeMenu()
      await invoke('quit_app').catch(console.error)
    }
    ```

    Critical implementation notes:
    - Use `@mousedown.prevent` (NOT `@click`) on menu items — `@click` fires after focus shift which triggers auto-hide before the click completes (Pitfall 1 from RESEARCH.md)

    - Use `position: fixed; inset: 0` on `.menu-backdrop` — covers the entire window regardless of scroll
    - Backdrop z-index: 99, menu z-index: 100 — menu sits above backdrop (Pitfall 4 from RESEARCH.md)
