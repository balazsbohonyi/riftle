pub struct SearchResult {
    pub id: String,
    pub name: String,
    pub icon_path: String,
    pub path: String,
    pub kind: String,   // "app" | "system"

const SYSTEM_COMMANDS: &[(&str, &str)] = &[

    ("system:shutdown", "Shutdown"),
    ("system:restart",  "Restart"),

RESEARCH.md key API facts:
- `Pattern::parse(query, CaseMatching::Ignore, Normalization::Smart)` — parse query
- `Utf32String::from(app.name.as_str())` — convert haystack
- `pattern.score(haystack.slice(..), &mut matcher)` — returns Option<u32>
- `match_tier` returns u8: 2=prefix, 1=acronym, 0=fuzzy
- Acronym only applies when `query.len() >= 2`
- Sort key: tier DESC, score DESC, launch_count DESC
- System command filtering: simple `.contains()` on name, case-insensitive

    3. `pub fn score_and_rank(query: &str, apps: &[AppRecord]) -> Vec<SearchResult>`:
       - Create Matcher::new(Config::DEFAULT)
       - Pattern::parse(query, CaseMatching::Ignore, Normalization::Smart)
       - q_lower = query.to_lowercase()
       - filter_map: score each app via pattern.score(Utf32String::from(app.name.as_str()).slice(..), &mut matcher)
       - collect into Vec<ScoredResult> with tier, score, app ref
       - sort_unstable_by tier DESC, score DESC, launch_count DESC
       - truncate(50)
       - map to Vec<SearchResult> (icon_path: unwrap_or("generic.png"), kind: "app")
