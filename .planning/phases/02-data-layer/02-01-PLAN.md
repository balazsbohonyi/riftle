use std::path::PathBuf;
use tauri::{AppHandle, Manager};

/// Returns the data directory for this install.
/// Portable mode: exe_dir/data (when launcher.portable exists adjacent to exe).
/// Installed mode: %APPDATA%\com.riftle.launcher\ (via Tauri path resolver).
/// Guarantees the directory exists via create_dir_all before returning.
pub fn data_dir(app: &AppHandle) -> PathBuf {
    let exe_path = std::env::current_exe()
        .expect("cannot resolve current exe path");
    let exe_dir = exe_path.parent()
        .expect("exe has no parent directory")
        .to_path_buf();

    data_dir_from_exe_dir(&exe_dir, app)
}

// Internal helper — takes exe_dir explicitly so unit tests can inject a tempdir.
pub fn data_dir_from_exe_dir(exe_dir: &PathBuf, app: &AppHandle) -> PathBuf {
    let dir = if exe_dir.join("launcher.portable").exists() {
        exe_dir.join("data")
    } else {
        app.path()
            .app_data_dir()
            .expect("cannot resolve app data dir")
    };

    std::fs::create_dir_all(&dir)
        .unwrap_or_else(|e| panic!("cannot create data dir {:?}: {}", dir, e));
    dir
}
```

#[cfg(test)]
mod tests {
    use super::*;
    use std::fs;

    #[test]
    fn test_portable_detection_returns_data_subdir() {
        // Simulate exe_dir with launcher.portable present
        let temp = std::env::temp_dir().join("riftle_paths_test");
        let _ = fs::remove_dir_all(&temp);
        fs::create_dir_all(&temp).unwrap();
        fs::write(temp.join("launcher.portable"), "").unwrap();

        // Test the portable branch logic directly (without AppHandle)
        let dir = if temp.join("launcher.portable").exists() {
            temp.join("data")
        } else {
            panic!("should have detected portable marker");
        };

        fs::create_dir_all(&dir).unwrap();
        assert!(dir.exists(), "data subdir should have been created");
        assert_eq!(dir, temp.join("data"));

        // Cleanup
        let _ = fs::remove_dir_all(&temp);
    }

    #[test]
    fn test_no_portable_marker_takes_installed_branch() {
        let temp = std::env::temp_dir().join("riftle_paths_test_no_portable");
        let _ = fs::remove_dir_all(&temp);
        fs::create_dir_all(&temp).unwrap();
        // No launcher.portable written

        let is_portable = temp.join("launcher.portable").exists();
        assert!(!is_portable, "no portable marker should mean installed mode");

        let _ = fs::remove_dir_all(&temp);
    }
}
```
